void QuickSort(vector<int> nums, int low, int high) {
	int temp = nums[low];
	int i = low, j = high;
	if (i >= j) {
		return;
	}
	while (i < j) {
		while (i < j&&nums[j] >= temp) {
			j--;
		}
		if (i < j) {
			nums[i] = nums[j];
			i++;
		}
		while (i<j&&nums[i]<=temp){
			i++;
		}
		if (i < j) {
			nums[j] = nums[i];
			j--;
		}
	}
	nums[i] = temp;
	QuickSort(nums, low, i - 1);
	QuickSort(nums, i + 1, high);
}
